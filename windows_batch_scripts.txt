@echo off
REM setup.bat - Quick setup script for Windows
REM Double-click to run or execute from Command Prompt

echo ================================================================
echo RAG System - Windows Setup
echo ================================================================
echo.

REM Check if Python is installed
python --version >nul 2>&1
if %errorlevel% neq 0 (
    echo [ERROR] Python is not installed or not in PATH
    echo Please install Python 3.8+ from https://www.python.org/downloads/
    echo Make sure to check "Add Python to PATH" during installation
    pause
    exit /b 1
)

echo [INFO] Python found
python --version
echo.

REM Run setup script
echo [INFO] Running setup wizard...
echo.
python setup_windows.py

if %errorlevel% neq 0 (
    echo.
    echo [ERROR] Setup failed. Please check the errors above.
    pause
    exit /b 1
)

echo.
echo ================================================================
echo Setup Complete!
echo ================================================================
echo.
echo Next steps:
echo 1. Edit .env file with your API keys (if using OpenAI)
echo 2. Run: extract_documents.bat (to process your PDFs/DOCX)
echo 3. Run: build_database.bat (to create vector database)
echo 4. Run: start_chatbot.bat (to launch the chatbot)
echo 5. Run: run_tests.bat (to test different models)
echo.
pause

REM =================================================================
REM Save this as: setup.bat
REM =================================================================

REM =================================================================
REM clean.bat - Clean up generated files
REM =================================================================
@echo off
echo ================================================================
echo Cleanup Utility
echo ================================================================
echo.
echo WARNING: This will delete:
echo - Vector database (faiss_course_db)
echo - Knowledge graph (knowledge_graph_db)
echo - Test results (test_results)
echo - Metrics (chatbot_metrics)
echo.
echo Your .env file and source code will NOT be deleted
echo.
set /p confirm="Are you sure? (y/n): "

if /i not "%confirm%"=="y" (
    echo Cancelled
    pause
    exit /b 0
)

echo.
echo Cleaning up...

if exist faiss_course_db (
    rmdir /s /q faiss_course_db
    echo [DELETED] faiss_course_db
)

if exist knowledge_graph_db (
    rmdir /s /q knowledge_graph_db
    echo [DELETED] knowledge_graph_db
)

if exist test_results (
    rmdir /s /q test_results
    echo [DELETED] test_results
)

if exist chatbot_metrics (
    rmdir /s /q chatbot_metrics
    echo [DELETED] chatbot_metrics
)

if exist __pycache__ (
    rmdir /s /q __pycache__
    echo [DELETED] __pycache__
)

echo.
echo [SUCCESS] Cleanup complete
echo.
pause

REM =================================================================
REM Save this as: clean.bat
REM =================================================================
